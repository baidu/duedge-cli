NAME
    create-trigger -

DESCRIPTION
    创建触发器

SYNOPSIS
      create-trigger
    --domain-name <string>
    --regular <string>
    [--description <string>]
    [--function-brn <string>]
    [--function-name <string>]
    [--version <string>]
    [--off <int>]
    [--client-request <bool>]
    [--client-response <bool>]

OPTIONS
    --domain-name (string)
        触发器绑定的域名

    --regular (string)
        节点调用函数计算时正则匹配条件

    --description (string)
        规则的描述信息

    --function-brn (string)
        函数的唯一性资源标志符，1-170字符

    --function-name (string)
        函数名

    --version (string)
        函数版本(function-brn或者function-name + version 必须至少有一个不为空)

    --off (int)
        开关，0-开，1-关

    --client-request (bool)
        client_request触发器是否绑定函数

    --client-response (bool)
        client_response触发器是否绑定函数

        
    注:--client-request，--client-response至少有一个不为空

OUTPUT
    参数名称 -> (参数类型)
        参数说明

    errors -> (list)
        错误信息

    request_uuid -> (string)
        请求uuid

    result -> (dict)
        资源详情

    success -> (bool)
        请求成功与否标志
